/**
 * Signature Verification Module
 * Verifies webhook signatures using HMAC-SHA256
 * Requirements: 2.1, 2.2, 2.3, 2.4, 2.5, 2.6
 */
 
import { createHmac } from 'crypto';
 
/**
 * Extract signature from X-Hub-Signature header
 * Removes the 'sha256=' prefix if present
 * Requirements: 2.2
 */
export function extractSignatureFromHeaders(headers: Record<string, string>): string | null {
  // Headers can be lowercase or mixed case
  const signatureHeader = Object.entries(headers).find(
    ([key]) => key.toLowerCase() === 'x-hub-signature'
  );
 
  if (!signatureHeader) {
    return null;
  }
 
  const signatureValue = signatureHeader[1];
 
  // Remove 'sha256=' prefix if present
  if (signatureValue.startsWith('sha256=')) {
    return signatureValue.substring(7);
  }
 
  return signatureValue;
}
 
/**
 * Compute HMAC-SHA256 signature for a payload
 * Requirements: 2.1
 */
export function computeSignature(payload: string, secret: string): string {
  const hmac = createHmac('sha256', secret);
  hmac.update(payload);
  return hmac.digest('hex');
}
 
/**
 * Verify signature using constant-time comparison
 * Prevents timing attacks by comparing all characters
 * Requirements: 2.3, 2.4, 2.6
 */
export function verifySignature(
  payload: string,
  receivedSignature: string,
  secret: string
): boolean {
  const computedSignature = computeSignature(payload, secret);
 
  // Constant-time comparison to prevent timing attacks
  return constantTimeCompare(computedSignature, receivedSignature);
}
 
/**
 * Constant-time string comparison
 * Compares all characters even if mismatch is found early
 * Requirements: 2.6
 */
function constantTimeCompare(a: string, b: string): boolean {
  // If lengths differ, still compare all characters to prevent timing attacks
  const minLength = Math.min(a.length, b.length);
  let mismatch = a.length !== b.length ? 1 : 0;
 
  for (let i = 0; i < minLength; i++) {
    mismatch |= a.charCodeAt(i) ^ b.charCodeAt(i);
  }
 
  return mismatch === 0;
}
 
/**
 * Minify JSON string by removing unnecessary whitespace
 * Used as fallback for signature verification
 */
function minifyJson(jsonString: string): string {
  try {
    // Parse and re-stringify to normalize JSON
    const parsed = JSON.parse(jsonString);
    return JSON.stringify(parsed);
  } catch {
    // If parsing fails, return original string
    return jsonString;
  }
}
 
/**
 * Validate webhook signature with minified JSON fallback
 * First attempts verification with original body
 * If that fails, attempts with minified JSON
 * Requirements: 2.1, 2.2, 2.3, 2.4, 2.5, 2.6
 */
export function validateWebhookSignature(
  headers: Record<string, string>,
  body: string,
  secret: string
): [boolean, string | null] {
  const receivedSignature = extractSignatureFromHeaders(headers);
 
  if (!receivedSignature) {
    return [false, 'No signature found in X-Hub-Signature header'];
  }
 
  // Try verification with original body
  if (verifySignature(body, receivedSignature, secret)) {
    return [true, null];
  }
 
  // Try verification with minified JSON
  const minifiedBody = minifyJson(body);
  if (minifiedBody !== body && verifySignature(minifiedBody, receivedSignature, secret)) {
    return [true, null];
  }
 
  return [false, 'Signature verification failed'];
}
